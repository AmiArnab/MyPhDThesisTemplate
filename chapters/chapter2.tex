\begin{algorithm}[!htb]
  \caption{Experiment $\Real^{\SSE}(\lambda,Q)$ for static SSE}
  \label{chap:preliminaries_and_background:algo_sec_static_real}
  \small
  \begin{algorithmic}[1]
    \Statex
    \Function{$\Real^{\SSE}$}{$\lambda,Q$}
        \State{$N\leftarrow\Adv(\lambda)$}
        \State{$\left(\sk,{\st}_0,\EDB_0\right)\leftarrow\setup(\lambda,N)$}
        \For{$k \gets 1\ \textrm{to}\ Q$}
            \State{$q_k\leftarrow\Adv(\lambda,\EDB_{k-1},\tau_1,\ldots,\tau_{k-1})$}
            \State{$({\st}_k,\EDB_k,\DB(q_k))\leftarrow \search(\sk,{\st}_{k-1},q_k;\EDB_{k-1})$}
            \State{$\tau_k$ denote the view of the adversary after the $k^{\textnormal{th}}$ query}
        \EndFor
        \State{$b\leftarrow\Adv(\lambda,\EDB_Q,\tau_1,\ldots,\tau_{Q})$}
        \State\Return{$b$}
    \EndFunction    
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}[!htb]
  \caption{Experiment $\Ideal^{\SSE}(\lambda,Q,\mathcal{L})$ for static SSE}
  \label{chap:preliminaries_and_background:algo_sec_static_ideal}
  \small
  \begin{algorithmic}[1]
    \Statex
    \Function{$\Ideal^{\SSE}$}{$\lambda,Q,\mathcal{L}$}
        \State{Parse the leakage function $\mathcal{L}$ as: $\mathcal{L} = \left(\mathcal{L}^{\setup},\mathcal{L}^{\search}\right)$.}
        \State{$N\leftarrow\Adv(\lambda)$}
        \State{$\left({\st}_{\Sim},\EDB_0\right)\leftarrow{{\sim}_{\setup}}(\mathcal{L}^{\setup}(\lambda,N))$}
        \For{$k \gets 1\ \textrm{to}\ Q$}
            \State{$q_k\leftarrow\Adv(\lambda,\EDB_{k-1},\tau_1,\ldots,\tau_{k-1})$}
            \State{$({\st}_{\Sim},\EDB_k,\tau_k)\leftarrow {{\sim}_{\search}}({\st}_{\Sim},\mathcal{L}^{\search}(q_k);\EDB_{k-1})$}
            \State{$\tau_k$ denote the view of the adversary after the $k^{\textnormal{th}}$ query}
        \EndFor
        \State{$b\leftarrow\Adv(\lambda,\EDB_Q,\tau_1,\ldots,\tau_{Q})$}
        \State\Return{$b$}
    \EndFunction    
  \end{algorithmic}
\end{algorithm}
